def all
  a = []
  @store.transaction(true) do
    @store.roots.each do |k|
      a.push(@store[k])
    end
  end
  a
end

def all_without_old
  all.select do |server|
    Time.now - server[:updated] < MAX_AGE
  end
end

def most_idle
  all_without_old.sort do |a, b|
    a[:load_avg] <=> b[:load_avg]
  end.first
end

def set(domain, load_avg)
  @store.transaction do
    @store[domain] = {
      :domain => domain,
      :load_avg => load_avg,
      :updated => Time.now
    }
  end
end
