ESCAPES =

tools.ietf.org/html/rfc4515 lists these exceptions from UTF1 charset for filters. All of the following must be escaped in any normal string using a single backslash ('') as escape.

{
  "\0" => '00', # NUL            = %x00 ; null character
  '*'  => '2A', # ASTERISK       = %x2A ; asterisk ("*")
  '('  => '28', # LPARENS        = %x28 ; left parenthesis ("(")
  ')'  => '29', # RPARENS        = %x29 ; right parenthesis (")")
  '\\' => '5C', # ESC            = %x5C ; esc (or backslash) ("\")
}
ESCAPE_RE =

Compiled character class regexp using the keys from the above hash.

Regexp.new(
  "[" +
  ESCAPES.keys.map { |e| Regexp.escape(e) }.join +

def self.convert(entry, all=false)
  h = {}

  entry.each do |k,v|
    if ob = @@ldap2json[k.to_s]
      h[ob[:attr]] = ob[:convert].call(v)
    elsif all
      h[k] = v
    end
  end

  return h
end

def self.escape(string)

def self.ldap_attr_conversion(ldap_name, json_name, &convert)
  @@ldap2json[ldap_name.to_s] = {
    :attr => json_name.to_s,
    :convert => convert || lambda { |v| Array(v).first }
  }
end

def filter(filter, attributes=nil)
  res = []

  @ldap_conn.search(
    ldap_base,
    LDAP::LDAP_SCOPE_SUBTREE,
    filter,
    attributes
  ) do |entry|
    res.push(entry.to_hash)
  end

  return res
end
